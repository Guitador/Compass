# Caminho do arquivo CSV
csv_file_path = r'C:\Users\guiha\OneDrive\Documentos\pessoal\serios\Trabalho\Compass\GitHub\Sprint_3\Exercícios\Parte_2\actors.csv'

# Variáveis para armazenar o ator/atriz com maior número de filmes
mais_filmes = 0
ator_com_mais_filmes = ""

# Ler o arquivo CSV manualmente
with open(csv_file_path, mode='r', encoding='utf-8') as file:
    lines = file.readlines()
    header = lines[0].strip().split(',')  # Dividir o cabeçalho por vírgulas
    for line in lines[6:]:
        # Dividir cada linha por vírgulas
        data = line.strip().split(',')
        # Extrair o número de filmes e o nome do ator
        num_filmes = int(data[header.index('Number of Movies')])
        ator = data[header.index('Actor')]
        # Verificar se o número de filmes do ator/atriz é maior do que o máximo atual
        if num_filmes > mais_filmes:
            mais_filmes = num_filmes
            ator_com_mais_filmes = ator

# Resultado
print(f"O ator/atriz com maior número de filmes é {ator_com_mais_filmes} com {mais_filmes} filmes.")
