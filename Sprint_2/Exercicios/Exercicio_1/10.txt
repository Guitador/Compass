SELECT 
    v.nmvdd AS vendedor, 
    SUM(vd.qtd * vd.vrunt) AS valor_total_vendas, 
    ROUND(SUM(vd.qtd * vd.vrunt) * v.perccomissao / 100, 2) AS comissao
FROM 
    TBVENDEDOR v
JOIN 
    TBVENDAS vd ON v.cdvdd = vd.cdvdd
WHERE 
    vd.status = 'Conclu√≠do'
GROUP BY 
    v.nmvdd, v.perccomissao
ORDER BY 
    comissao DESC;
